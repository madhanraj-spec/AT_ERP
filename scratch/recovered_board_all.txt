5632: function getLocalDateOnly(isoString) {
5633:   if (!isoString) return '';
5634:   const d = new Date(isoString);
5635:   if (isNaN(d.getTime())) return '';
5636:   const y = d.getFullYear();
5637:   const m = String(d.getMonth() + 1).padStart(2, '0');
5638:   const day = String(d.getDate()).padStart(2, '0');
5639:   return `${y}-${m}-${day}`;
5640: }
